// --- เพิ่มโค้ดส่วนนี้เพื่อตรวจสอบการล็อกอินค้างไว้ --- import { onAuthStateChanged } from "https://www.gstatic.com/firebasejs/10.8.1/firebase-auth.js"; onAuthStateChanged(auth, (user) => { if (user) { // ถ้าผู้ใช้เคยล็อกอินอยู่แล้ว และยืนยันอีเมลแล้ว if (user.emailVerified) { console.log("พบผู้ใช้เดิม: ", user.email); // ส่งไปหน้า categories.html ทันที (ไม่ต้องผ่านหน้า Login) window.location.href = 'categories.html'; } } else { // ถ้าไม่มีผู้ใช้ล็อกอินอยู่ ก็ปล่อยให้แสดงหน้า Login ตามปกติ console.log("ยังไม่ได้ล็อกอิน"); } }); // ------------------------------------------
