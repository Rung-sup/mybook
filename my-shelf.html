function renderShelf(uid) { const shelfDiv = document.getElementById('shelf-list'); const history = JSON.parse(localStorage.getItem(`puk_history_${uid}`)) || {}; const bookIds = Object.keys(history); // --- กรณีผู้ใช้ใหม่ (ยังไม่มีประวัติการอ่าน) --- if (bookIds.length === 0) { shelfDiv.innerHTML = `
👋

ยินดีต้อนรับคุณเข้าสู่คลังหนังสือ

ดูเหมือนว่าคุณยังไม่ได้เริ่มอ่านเล่มไหนเลย
ลองไปเลือกหนังสือที่น่าสนใจในหอสมุดดูไหมครับ?

สำรวจหอสมุดตอนนี้ ❯
`; // ซ่อนหัวข้อ "อ่านค้างไว้" และปุ่มด้านล่างเพื่อไม่ให้ดูรก document.querySelector('.section-title').style.display = 'none'; document.querySelector('.container > div:last-child').style.display = 'none'; return; } // --- กรณีผู้ใช้เดิม (มีประวัติการอ่าน) --- document.querySelector('.section-title').style.display = 'flex'; document.querySelector('.container > div:last-child').style.display = 'block'; bookIds.sort((a, b) => history[b].time - history[a].time); shelfDiv.innerHTML = ''; bookIds.forEach(id => { const data = history[id]; const card = document.createElement('div'); card.className = 'book-card'; card.innerHTML = `
📕
${data.s === 'อ่านจบแล้ว' ? 'จบแล้ว' : 'หน้า ' + data.p}
${data.title}
`; const bookUrl = atob(id); card.onclick = () => { const parts = bookUrl.split('/'); const cat = parts[parts.length - 3] || '1_PetchPraUma'; window.location.href = `register.html?cat=${cat}&name=${encodeURIComponent(data.title)}`; }; shelfDiv.appendChild(card); }); }
